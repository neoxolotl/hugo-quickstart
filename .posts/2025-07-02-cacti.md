version: '2'
services:
  cacti:
    image: "mrlesmithjr/cacti:latest"
    depends_on:
      - "db"
    links:
      - "db"

      CACTI_DB: "cactidb"

      CACTI_USER_PASSWORD: "cacti"

    image: "mrlesmithjr/mysql:latest"
    volumes:
      - "db:/var/lib/mysql"
    restart: "always"

      MYSQL_ROOT_PASSWORD: "cacti"
      MYSQL_DATABASE: "cactidb"
